Code Snippet

            const int sensorOutputPin = 11;  // sensorPin output
            const int sensorInputPin = 10;  // sensorPin input
            const int ledPin = 9;  // PWM pin controlling the MOSFET 
            #define SOUND_SPEED 0.0343  // Speed of sound in air (343 m/s) converted to cm/us
            
            void setup() { //setup code
                pinMode(sensorOutputPin, OUTPUT); // Set sensor pin as an output to send the pulse
                pinMode(sensorInputPin, INPUT); // Set sensor pin as an input to receive pulse
                pinMode(ledPin, OUTPUT);  // Set ledPin as an output
            }
            
            void loop() { //repeat code 
                digitalWrite(sensorOutputPin, LOW);  // start with a low pulse
                delayMicroseconds(2); // start with a low pulse for two seconds
                digitalWrite(sensorOutputPin, HIGH);  // send high pulse for echo
                delayMicroseconds(10); // send high pulse for echo for ten seconds
                digitalWrite(sensorOutputPin, LOW); // end with a low pulse for two seconds
            
                
                long duration = pulseIn(sensorInputPin, HIGH); // read the time it takes for the echo to return
            
                
                int distance = duration * SOUND_SPEED / 2; // calculate distance in centimeters
            
            
                int brightness = map(constrain(distance, 10, 100), 10, 100, 255, 0);  // map distance (10cm to 100cm) to LED brightness (0-255)
            
                analogWrite(ledPin, brightness); // set the brightness using analogWrite()
            
                delay(50); // delay to stabilize readings
            }
